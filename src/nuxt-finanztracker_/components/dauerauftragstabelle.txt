<template>
  <div class="card">
    <!-- Header -->
    <div class="flex justify-between items-center mb-4">
      <h2 class="text-lg font-semibold text-brand-600 dark:text-brand-300">
        Übersicht der Daueraufträge
      </h2>

      <input
        type="text"
        v-model="search"
        placeholder="Suchen..."
        class="form-input w-1/3 dark:border-gray-500 dark:bg-gray-800"
      />
    </div>

    <!-- Tabelle -->
    <div class="table-container">
      <table class="table">
        <thead>
          <tr>
            <th>Name</th>
            <th>Kategorie</th>
            <th class="text-right">Betrag</th>
            <th>Intervall</th>
            <th class="text-center">Aktionen</th>
          </tr>
        </thead>

        <tbody>
          <tr
            v-for="auftrag in filteredAuftraege"
            :key="auftrag.id"
            class="hover:bg-gray-100 dark:hover:bg-gray-700 transition-colors"
          >
            <td>{{ auftrag.name }}</td>
            <td>{{ auftrag.kategorie }}</td>

            <td
              class="text-right font-medium"
              :class="auftrag.betrag < 0
                ? 'text-red-500 dark:text-red-400'
                : 'text-teal-600 dark:text-teal-400'"
            >
              {{ auftrag.betrag.toFixed(2) }} €
            </td>

            <td>{{ auftrag.intervall }}</td>

            <td class="text-center space-x-4">
              <button
                class="text-teal-600 hover:text-teal-400 dark:text-teal-400 dark:hover:text-teal-300"
                @click="$emit('edit', auftrag)"
              >
                <i class="fas fa-pen"></i>
              </button>

              <button
                class="text-gray-400 hover:text-red-500"
                @click="$emit('delete', auftrag)"
              >
                <i class="fas fa-trash"></i>
              </button>
            </td>
          </tr>
        </tbody>
      </table>
    </div>
  </div>
</template>

<script setup>
import { ref, computed } from "vue";

const props = defineProps({
  auftraege: { type: Array, required: true }
});

const search = ref("");

const filteredAuftraege = computed(() => {
  return props.auftraege.filter(a =>
    Object.values(a).some(v =>
      String(v).toLowerCase().includes(search.value.toLowerCase())
    )
  );
});
</script>
